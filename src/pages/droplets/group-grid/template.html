Initializing Plugin

This droplet uses the Filter Grid plugin. View that page for further documentation. Below describes the code needed for this droplet's particular "output" markup. 

$('.drzFilter-grid').drzFilterGrid({
  feed: [
    {
      created: '2020-06-25T19:31:14.835Z',
      description: 'Group description.',
      thumbnail: '/static/images/mock-group.svg',
      alias: 'Group Name',
      pageLink: '#',
      customLink: '#',
      categories: ['red group'],
    },
    ...
  ],
  gridContainer: 'drzGroup-grid-container',
  sortKey: 'created',
  assets: 'bg',
  output(data) {
    return `
      <div class="drzGroup-grid-card">
        <a class="drzGroup-grid-cardInner" href="#">
          <div
            class="drzGroup-grid-cardImage"
            data-asset="${data.image}">
          </div>
          <div class="drzGroup-grid-cardInfo">
            <span class="drzGroup-grid-cardTitle">${data.title}</span>
          </div>
        </a>
      </div>
    `;
  },
});
