Initializing Plugin

This droplet uses the Filter Grid plugin. View that page for further documentation. Below describes the code needed for this droplet's particular "output" markup. 

$('.drzFilter-grid').drzFilterGrid({
  feed: [
    {
      description: 'Some Description',
      image: '/static/images/mock-place.svg',
      name: 'Some House',
      address: {
        value: 'Scranton, PA',
        lat: 32.0298602,
        lng: -102.1075392,
      },
      mapPin: 'https://cdn.drzzle.app/mock-images/icons/maps-default-pin.svg',
      pageLink: '#',
      categories: ['house'],
    },
  ],
  gridContainer: 'drzPlace-items-container',
  // if you want a map shown
  map: {
    title: 'name',
  },
  output(data) {
    return `
      <li class="drzPlace-item-card">
        <a href="${data.pageLink}" class="drzPlace-item-link">
          <div class="drzPlace-item-imgWrap">
            <img class="drzPlace-item-img" data-asset="${data.image}" alt="${data.name}" />
          </div>
          <div class="drzPlace-item-data">
            <div class="drzPlace-item-top">
              <h4 class="drzPlace-item-header">${data.name}</h4>
            </div>
            <hr class="drzPlace-item-break" />
            <p class="drzPlace-item-preview">
              ${data.description}
            </p>
          </div>
        </a>
      </li>
    `;
  },
});
